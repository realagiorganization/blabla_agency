\BilingualFrame
  {Cadence / Cadence}
  {
    \begin{itemize}
      \item client sends a draft technical specification
      \item the draft is discussed at length in calls, video calls, chats, comments, and reviews
      \item AI generates and deploys a demo or staging version
      \item feedback is collected on the real demo
      \item the loop repeats
    \end{itemize}
  }
  {
    \begin{itemize}
      \item клиент присылает черновую техническую спецификацию
      \item черновик подробно обсуждается в звонках, видеозвонках, чатах, комментариях и ревью
      \item AI генерирует и деплоит demo- или staging-версию
      \item фидбек собирается по реальному демо
      \item цикл повторяется
    \end{itemize}
  }
